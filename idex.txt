    <script>
        const feriadosEstaduais = ["2026-01-01", "2026-02-16", "2026-02-17", "2026-02-18", "2026-03-06", "2026-04-03", "2026-04-21", "2026-05-01", "2026-06-04", "2026-06-24", "2026-09-07", "2026-10-12", "2026-10-28", "2026-11-02", "2026-11-15", "2026-12-08", "2026-12-25"];

        const feriadosMunicipais = {
            recife: ["2026-07-16"], // N. Sra do Carmo
            jaboatao: ["2026-01-15", "2026-05-04"], // Santo Amaro e Emancipação
            olinda: ["2026-08-06", "2026-11-10"], // S. Salvador do Mundo e Emancipação
            paulista: ["2026-09-04", "2026-12-15"], // Emancipação e Padroeira
            cabo: ["2026-07-09"], // Emancipação
            ipojuca: ["2026-03-30"] // Emancipação
        };

        function eDiaUtil(data, cidade) {
            const diaSemana = data.getUTCDay();
            const stringData = data.toISOString().split('T')[0];
            const municipais = feriadosMunicipais[cidade] || [];
            return diaSemana !== 0 && diaSemana !== 6 && !feriadosEstaduais.includes(stringData) && !municipais.includes(stringData);
        }

        function calcularPrazo() {
            const dataInput = document.getElementById('dataInicio').value;
            const cidade = document.getElementById('cidade').value;
            const tipo = document.getElementById('tipoPrazo').value;
            const diasParaSomar = parseInt(document.getElementById('qtdDias').value);

            if (!dataInput) { alert("Por favor, escolha o dia que você recebeu o documento."); return; }

            let dataAtual = new Date(dataInput);
            let diasContados = 0;

            // Início da contagem: Primeiro dia útil após o recebimento
            dataAtual.setUTCDate(dataAtual.getUTCDate() + 1);
            while (!eDiaUtil(dataAtual, cidade)) {
                dataAtual.setUTCDate(dataAtual.getUTCDate() + 1);
            }

            if (tipo === 'civil') {
                while (diasContados < (diasParaSomar - 1)) {
                    dataAtual.setUTCDate(dataAtual.getUTCDate() + 1);
                    if (eDiaUtil(dataAtual, cidade)) {
                        diasContados++;
                    }
                }
            } else {
                dataAtual.setUTCDate(dataAtual.getUTCDate() + diasParaSomar - 1);
                while (!eDiaUtil(dataAtual, cidade)) {
                    dataAtual.setUTCDate(dataAtual.getUTCDate() + 1);
                }
            }

            const opcaoData = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
            document.getElementById('dataFinalTexto').innerText = dataAtual.toLocaleDateString('pt-BR', { ...opcaoData, timeZone: 'UTC' });
            document.getElementById('resultado').classList.remove('hidden');
        }
    </script>
